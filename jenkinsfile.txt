pipeline {
  environment {
    registry = "https://hub.docker.com/repository/docker/kapilnegi/java-maven"
    registryCredential = 'dockerhub'
    dockerImage = ''
  }


    agent any
    tools {
        maven 'apache-maven-3.6.3' 
    }
    stages {
      
    stage('Push image') {
       steps{
        script {
          docker.withRegistry( '', registryCredential ) {
            dockerImage.push()
          }
        }
      }

     
    }

  }

}
